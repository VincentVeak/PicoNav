<script>
document.addEventListener('DOMContentLoaded', function() {
  initGearTable('headgear');
  initGearTable('bodygear');
});
function initGearTable(type) {
  const searchInput = document.getElementById(type + '-search');
  const table = document.getElementById(type + '-table');
  if (!searchInput || !table) return;
  searchInput.addEventListener('input', function() { filterGearTable(type); });
  const headers = table.querySelectorAll('thead th.sortable');
  headers.forEach((header, index) => { header.addEventListener('click', function() { sortGearTable(type, index); }); });
  updateGearCount(type);
}
function filterGearTable(type) {
  const searchInput = document.getElementById(type + '-search');
  const table = document.getElementById(type + '-table');
  const searchTerm = searchInput.value.toLowerCase();
  const rows = table.getElementsByTagName('tbody')[0].getElementsByTagName('tr');
  let visibleCount = 0;
  for (let row of rows) {
    const name = row.cells[2].textContent.toLowerCase();
    const obtain = row.cells[3].textContent.toLowerCase();
    if (name.includes(searchTerm) || obtain.includes(searchTerm)) { row.classList.remove('hidden-row'); visibleCount++; }
    else { row.classList.add('hidden-row'); }
  }
  document.getElementById('showing-' + type).textContent = visibleCount;
}
function sortGearTable(type, columnIndex) {
  const table = document.getElementById(type + '-table');
  const tbody = table.getElementsByTagName('tbody')[0];
  const rows = Array.from(tbody.getElementsByTagName('tr'));
  const header = table.querySelectorAll('thead th')[columnIndex];
  const currentDirection = header.dataset.sortDirection || 'none';
  const direction = currentDirection === 'asc' ? 'desc' : 'asc';
  header.dataset.sortDirection = direction;
  rows.sort((a, b) => {
    let aValue = a.cells[columnIndex].textContent.trim();
    let bValue = b.cells[columnIndex].textContent.trim();
    return direction === 'asc' ? aValue.localeCompare(bValue) : bValue.localeCompare(aValue);
  });
  rows.forEach(row => tbody.appendChild(row));
  updateSortIndicators(table, columnIndex, direction);
}
function updateSortIndicators(table, activeColumn, direction) {
  const headers = table.querySelectorAll('thead th');
  headers.forEach((header, index) => {
    const arrow = header.querySelector('.sort-arrow');
    if (arrow) {
      if (index === activeColumn) { header.classList.add('sorted'); arrow.textContent = direction === 'asc' ? 'â†‘' : 'â†“'; }
      else { header.classList.remove('sorted'); arrow.textContent = 'â‡…'; }
    }
  });
}
function updateGearCount(type) {
  const table = document.getElementById(type + '-table');
  const totalRows = table.getElementsByTagName('tbody')[0].getElementsByTagName('tr').length;
  document.getElementById('total-' + type).textContent = totalRows;
  document.getElementById('showing-' + type).textContent = totalRows;
}
